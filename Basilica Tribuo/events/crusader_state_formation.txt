character_event = {
	id = 666710
	desc = "Who shall take the throne of Jerusalem?"

	is_triggered_only = yes
	
	option = {
		name = "A dear family member shall take the throne..."
        trigger = {
            any_dynasty_member = {
                father = {
                    NOT = { has_landed_title = c_jerusalem }
                }
                mother = {
                    NOT = { has_landed_title = c_jerusalem }
                }
                age = 16
                is_female = no
                NOT = { demesne_size = 1 }
            }
        }
        random_dynasty_member = {
			limit = {
				father = {
                    NOT = { has_landed_title = c_jerusalem }
                }
                mother = {
                    NOT = { has_landed_title = c_jerusalem }
                }
                age = 16
                is_female = no
                NOT = { demesne_size = 1 }
			}
			c_jerusalem = { gain_title = PREV }
            k_jerusalem = { gain_title = PREV }
		}
    }
    option = {
		name = "A dear family member who is a vassal shall take the throne..."
        trigger = {
            any_vassal = {
                dynasty = ROOT
                age = 16
                is_female = no
            }
        }
        random_vassal = {
			limit = {
                dynasty = ROOT
                age = 16
                is_female = no
			}
            character_event = { id = 666713 }
		}
    }
    
    option = {
		name = "I can not manage the kingdom. We must find a member of my house who can..."
        trigger = {
            NOT = {
                any_dynasty_member = {
                    father = {
                        NOT = { has_landed_title = c_jerusalem }
                    }
                    mother = {
                        NOT = { has_landed_title = c_jerusalem }
                    }
                    age = 16
                    is_female = no
                }
            }
        }
        create_character = {
                random_traits = yes
                female = no
                age = 16
		}
        new_character = {
                c_jerusalem = { gain_title = PREV }
                k_jerusalem = { gain_title = PREV }
            }        
    }
    option = {
		name = "I will return my land to my liege and I will take the throne."
        trigger = {
            independent = no
        }
        top_liege = {
            character_event = { id = 666711 }
        }
    }
    option = {
		name = "I will hand the realm to a trusted vassal and I will take the throne."
        trigger = {
            independent = yes
            tier = king
        }
        random_vassal = {
			limit = {
                is_feudal = yes
                tier = duke
			}
			character_event = { id = 666712 }
		}
    }
    option = {
		name = "I will hand the realm to a trusted vassal and I will take the throne."
        trigger = {
            independent = yes
            tier = duke
        }
        random_vassal = {
			limit = {
                is_feudal = yes
                tier = count
			}
			character_event = { id = 666712 }
		}
    }
    option = {
		name = "I will hand the realm to a trusted vassal and I will take the throne."
        trigger = {
            independent = yes
            tier = count
        }
        random_vassal = {
			limit = {
                tier = baron
			}
			character_event = { id = 666712 }
		}
    }
}
